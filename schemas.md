# COVID Database Schemas

## Table: COVIDDEATHS

```sql
CREATE TABLE COVID_DATABASE.PUBLIC.COVIDDEATHS (
    ISO_CODE VARCHAR(16777216),
    CONTINENT VARCHAR(16777216),
    LOCATION VARCHAR(16777216),
    DATE DATE,
    TOTAL_CASES NUMBER(38,0),
    NEW_CASES NUMBER(38,0),
    NEW_CASES_SMOOTHED NUMBER(38,0),
    TOTAL_DEATHS NUMBER(38,0),
    NEW_DEATHS NUMBER(38,0),
    NEW_DEATHS_SMOOTHED NUMBER(38,0),
    TOTAL_CASES_PER_MILLION NUMBER(38,0),
    NEW_CASES_PER_MILLION NUMBER(38,0),
    NEW_CASES_SMOOTHED_PER_MILLION NUMBER(38,0),
    TOTAL_DEATHS_PER_MILLION NUMBER(38,0),
    NEW_DEATHS_PER_MILLION NUMBER(38,0),
    NEW_DEATHS_SMOOTHED_PER_MILLION NUMBER(38,0),
    HOSP_PATIENTS VARCHAR(16777216),
    HOSP_PATIENTS_PER_MILLION VARCHAR(16777216),
    WEEKLY_ICU_ADMISSIONS VARCHAR(16777216),
    WEEKLY_ICU_ADMISSIONS_PER_MILLION VARCHAR(16777216),
    WEEKLY_HOSP_ADMISSIONS VARCHAR(16777216),
    WEEKLY_HOSP_ADMISSIONS_PER_MILLION VARCHAR(16777216),
    POPULATION NUMBER(38,0)
);
```
